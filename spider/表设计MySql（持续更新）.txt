USER_ORIGIN
/*
	用户订阅的源
*/
USER_ID,ORIGIN_ID,

------------------------------------------------------------------------------------

BD_FAVOR
/*
	收藏类别
*/
USER_ID,FAVOR_ID,NAME

------------------------------------------------------------------------------------

FAVOR_POST
/*
	用户收藏的文章，如果 FAVOR_ID 为空，则为待读
*/
USER_ID,POST_ID,FAVOR_ID,

------------------------------------------------------------------------------------

RECOMMEND_POST
/*
	系统推荐给用户的文章
*/
USER_ID,POST_ID

------------------------------------------------------------------------------------

POST
/*
	文章
*/
POST_ID, URL, DIGEST, THUMB

------------------------------------------------------------------------------------

POST_DETAIL
/*
	文章详细内容
	分离开详细内容，主要是详细内容栏位太大，如果需要显示文章列表，完全不用查询详细内容
*/
POST_ID,DETAIL

------------------------------------------------------------------------------------

POST_SUBJECT
/*
	文章主题
	文章可能同时包含几个主题
*/
POST_ID,SUBJECT_ID

------------------------------------------------------------------------------------

BD_ORIGIN
/*
	订阅源
	这里定义为源，可能一个站点，有几个频道，需要分开源。
	这张表的作用，是用来提供一个源列表，用户可以订阅其中的源，订阅之后，会在推荐中显示订阅源的文章。
*/
ORIGIN_ID,ORIGIN_NAME,

------------------------------------------------------------------------------------

BD_SUBJECT
/*
	订阅主题
	定义一些主题，用户可以订阅这些主题，订阅之后，在推荐中显示订阅主题的文章
*/
SUBJECT_ID,NAME,
